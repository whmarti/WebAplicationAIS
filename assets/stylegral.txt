public class UserDA
    {
        private String strConn = ConfigurationManager.AppSettings["DataBase"].ToString();
        private  String EncryptClient = ConfigurationManager.AppSettings["EncryptClient"];
        private  String Passkey = ConfigurationManager.AppSettings["Passkey"];
        

            /// <summary>
            /// Request for User table information to the Data Base.
            /// </summary>
            /// <param name="pMError"></param>
            /// <returns></returns>
            public List<User> getUsers( ref String mError )
        {
            List<User> _users = null;
            SqlDataReader _rstReader = null;
            string _sql = "select * from UserMega";

            try
            {
                using (SqlConnection _conn = new SqlConnection(strConn))
                {
                    SqlCommand cmd = new SqlCommand(_sql, _conn);
                    _conn.Open();
                    _rstReader = cmd.ExecuteReader();
                    _users = new List<User>();
                    while (_rstReader.Read())
                    {

                        Assignment.App_Code.User _newUser = new Assignment.App_Code.User();
                        _newUser.IdUser = 0;
                        _newUser.name = "";
                        _newUser.userId = "";
                        _newUser.pass = "";

                        _newUser.IdUser = Convert.ToInt32(_rstReader["IdUser"]);
                        _newUser.userId = Convert.ToString(_rstReader["userId"]).Trim();
                        _newUser.name = Convert.ToString(_rstReader["name"]).Trim();
                        //_newUser.pass = Convert.ToString(_rstReader["pass"]).Trim();      
                        _users.Add(_newUser);

                    }
                    _rstReader.Close();
                }
            }
            catch (Exception ex)
            {
                mError = ex.Message;
            }
            return _users;
        }


   
        /// <summary>
        /// Function to authenticate the credentials of a user who intends to enter the system.
        /// </summary>
        /// <param name="pUser"></param>
        /// <param name="pPass"></param>
        /// <param name="pMError"></param>
        /// <returns></returns>
        public User getUserLoginBL(String pUser, String pPass, ref String mError  )
        {
            Assignment.App_Code.User _user = null;
            SqlDataReader _rstReader = null;
            string _sql = "select * from UserMega where userId=@User and pass=HASHBYTES('SHA2_512', @pass+CAST(@salt AS NVARCHAR(36))) ";

            try
            {
                using (SqlConnection _conn = new SqlConnection(strConn))
                {
                    SqlCommand _cmd = new SqlCommand(_sql, _conn);
                    _cmd.Parameters.AddWithValue("@User", pUser);
                    _cmd.Parameters.AddWithValue("@pass", pPass);
                    _cmd.Parameters.AddWithValue("@salt", EncryptClient);
                    _conn.Open();
                    _rstReader = _cmd.ExecuteReader();
                    if (_rstReader.Read())
                    {

                        _user = new Assignment.App_Code.User();
                        _user.IdUser = Convert.ToInt32(_rstReader["IdUser"]);
                        _user.userId = Convert.ToString(_rstReader["userId"]).Trim();
                        _user.name = Convert.ToString(_rstReader["name"]).Trim();
                        mError = "0";
                        //_user.pass = Convert.ToString(_rstReader["pass"]).Trim();
                        // _user.date = Convert.ToString(_rstReader["date"]).Trim();

                    }
                    else
                    {
                        mError ="Incorrect credentials, try again.";
                    }
